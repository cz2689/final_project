mari_df = nsduh_df %>%
  filter(mjrec == c("1", "2", "3")) %>%
  mutate(mjrec = recode(mjrec, "1" = "<= 30 days",
                        "2" = "1-12 months",
                        "3" = ">12 months")) %>%
  group_by(mjrec)
